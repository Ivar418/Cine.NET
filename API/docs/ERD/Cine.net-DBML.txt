Table movies {
  id int [pk, increment]
  title longtext [not null]
  tmdb_id int [not null, note: 'TMDB reference — primary source of movie metadata']
  information_language longtext [not null]
  language longtext
  poster_url longtext
  backdrop_url longtext
  youtube_trailer_key longtext
  runtime int
  imdb_id longtext
  release_date longtext
  about longtext
  age_indication longtext
  spoken_language_name longtext
  spoken_language_code_iso6391 longtext
  genres_ids longtext [note: 'Stores TMDB genre IDs as text; no FK to local genres table']
  row_created_timestamp_utc longtext [not null]
  row_updated_timestamp_utc longtext
  row_deleted_timestamp_utc longtext
}

Table genres {
  id int [pk, increment]
  tmdb_id int [not null, note: 'TMDB genre ID']
  name longtext [not null]
  language longtext [not null, note: 'Multiple languages cached (e.g. en, nl). (tmdb_id, language) should be unique']
}

Table auditoriums {
  Id int [pk, increment]
  Name longtext [not null]
  RowConfigJson longtext [not null, note: 'Seat layout stored as JSON']
}

Table showings {
  Id int [pk, increment]
  MovieId int [not null, ref: > movies.id]
  AuditoriumId int [not null, ref: > auditoriums.Id]
  IsThreeD boolean [not null]
  StartsAt longtext [not null]
  AuditoriumLayoutSnapshot longtext [not null, note: 'Snapshot of auditorium layout at time of showing']
}

Table reservations {
  Id char(36) [pk, note: 'UUID']
  ShowingId int [not null, ref: > showings.Id]
  SeatsJson longtext [not null, note: 'Selected seats stored as JSON']
  Status longtext [not null]
  CreatedAt datetime [not null]
}

Table ticket_types {
  id int [pk, increment]
  name longtext [not null, note: 'Seeded on startup (e.g. Adult, Child, Student)']
  discount decimal [not null, note: 'Discount applied to base ticket price']
}

Table tickets {
  id int [pk, increment]
  showing_id int [not null, ref: > showings.Id]
  shown_date_time_utc longtext [not null]
  seat_number longtext [not null]
  ticket_type longtext [not null, note: 'Stores ticket type name as text — no FK to ticket_types; consider referencing ticket_types.id instead']
  price decimal [not null]
  status longtext [not null]
  payment_status longtext [not null]
  qr_code_guid longtext
  qr_is_active boolean [not null]
  purchase_date datetime [not null]
  row_created_timestamp_utc longtext [not null]
  row_updated_timestamp_utc longtext
  row_deleted_timestamp_utc longtext
}

Table pricing_options {
  id int [pk, increment]
  name longtext [not null]
  price_modifier decimal [not null]
}

Table pricing_configs {
  id int [pk, increment]
  key longtext [not null]
  value decimal [not null]
}

Table payment_methods {
  Code varchar(20) [pk]
  DisplayName varchar(50) [not null]
}

Table orders {
  Id int [pk, increment]
  OrderCode longtext [not null]
  CreatedAtUtc datetime [not null]
  TotalAmount decimal [not null]
  OrderType longtext [not null]
  PaymentStatus longtext [not null]
  PaymentMethod varchar(20) [not null, ref: > payment_methods.Code]
  CashierEmployeeId int [note: 'FK to employees table — not yet implemented']
  IsPrinted boolean [not null]
}

Table order_tickets {
  OrderId int [not null, ref: > orders.Id]
  TicketId int [not null, ref: > tickets.id]

  indexes {
    (OrderId, TicketId) [pk]
  }
}

Table email_subscriptions {
  id int [pk, increment]
  email longtext [not null]
}
